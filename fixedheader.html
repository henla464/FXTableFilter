Description

.FixedHeaderTable is a JQuery plugin for adding a fixed header to any HTML table. It can easily be used with FXTableFilter. The only thing we need to adjust is the height of the wrapper div that the .FixedHeaderTable creates so that it can accomodate the filter row.

           $(document).ready(function () {
             var tableHeightWithoutFilters = 400;
             $("#demotable-fixedheader").fixedHeaderTable({ footer: true, cloneHeadToFoot: true, fixedColumn: false, height: tableHeightWithoutFilters });

             var tooltip = 'Your tooltip here!';
             // Initialise Plugin
             var optionsfixedheader = {
               columnFilters: [ 	
                 { columnName: 'Model', inputType: 'text', toolTipMessage: tooltip},
                 { columnName: 'YearFrom', inputType: 'text', toolTipMessage: tooltip},
                 { columnName: 'YearTo', inputType: 'text', toolTipMessage: tooltip}
               ]
             };

             $('.fht-table-wrapper').css({ height: tableHeightWithoutFilters + 30 });
             $('#demotable-fixedheader').tableFilter(optionsfixedheader);
           });
