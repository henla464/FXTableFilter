Description

The Yes/No column has a dropdown list instead of a textbox in the filter row. When the filter type is set to 'ddl' then a dropdown list populated with all distinct values in the column will be used instead of a textbox

Code

  • JavaScript
  • HTML
  • CSS
           $(document).ready(function () {
             var tooltip = 'Your tooltip here!';
             // Initialise Plugin
             var optionsbigtable = {
               columnFilters: [ 	
                  { columnName: 'col1', inputType: 'text', toolTipMessage: tooltip},
                  { columnName: 'col2', inputType: 'text', toolTipMessage: tooltip},
                  { columnName: 'col3', inputType: 'ddl'},
                  { columnName: 'col4', inputType: 'text'}
               ]
             };
 
             $('#demotable-bigtable').tableFilter(optionsbigtable);
           });
