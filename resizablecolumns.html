Description

This example shows that FXTableFilter works fine together with the resizable columns plugin ( https://github.com/dobtco/jquery-resizable-columns )

Code

  • JavaScript
  • HTML
  • CSS
           $(document).ready(function () {
             var tooltip = 'Your tooltip here!';
             var optionsresizablecolumns = {
            	columnFilters: [ 	
            				{ columnName: 'Model', inputType: 'text', toolTipMessage: tooltip},
            			 	{ columnName: 'YearFrom', inputType: 'text', toolTipMessage: tooltip},
            				{ columnName: 'YearTo', inputType: 'text', toolTipMessage: tooltip}
            			]
             };

             $('#demotable-resizablecolumns').tableFilter(optionsresizablecolumns);
             $("#demotable-resizablecolumns").resizableColumns();
           });
